39 float posY = (rand() % (mapSize - 1)) + 0;
40 float dirX = 0;
41 float dirY = 0;

48 // droplet position bound to cell
49 int nodeX = (int)posX;
50 int nodeY = (int)posY;
51 int dropletIndex = nodeY * mapSize + nodeX;
52 // calculate droplet's offset inside the cell (0,0) = at NW node, (1,1) = at SE node
53 float cellOffsetX = posX - (float)nodeX;
54 float cellOffsetY = posY - (float)nodeY;
55
56 // calculate droplet's height and direction of flow with bilinear interpolation of surrounding heights
57 HeightAndGradient heightAndGradient = CalculateHeightAndGradient(mapData, mapSize, posX, posY);
58
59 // update the droplet's direction and position (move position 1 unit regardless of speed)
60 dirX = (dirX * inertia - heightAndGradient.gradientX * (1 - inertia)); // lerp with old dir by using inertia as mix value
61 dirY = (dirY * inertia - heightAndGradient.gradientY * (1 - inertia));
62
63 // normalize direction
64 float len = sqrtf(dirX * dirX + dirY * dirY);
65 if (len > 0.0001f)
66 {
67 dirX /= len;
68 dirY /= len;
69 }
70 // update droplet position based on direction (move 1 unit)
71 posX += dirX;
72 posY += dirY;
73
74 // stop simulating droplet if it's not moving or has flowed over edge of map
75 if ((dirX == 0 && dirY == 0) || posX < 0 || posX >= mapSize - 1 || posY < 0 || posY >= mapSize - 1)
76 {
77 break;
78 }
79
80 // find the droplet's new height and calculate the deltaHeight
81 float newHeight = CalculateHeightAndGradient(mapData, mapSize, posX, posY).height;
82 float deltaHeight = newHeight - heightAndGradient.height;
83
84 // calculate the droplet's sediment capacity (higher when moving fast down a slope and contains lots of water)
85 float sedimentCapacity = std::max(-deltaHeight * speed * water * sedimentCapacityFactor, minSedimentCapacity);
86
87 // if carrying more sediment than capacity, or if flowing uphill:
88 if (sediment > sedimentCapacity || deltaHeight > 0)
89 {
90 // DEPOSIT
91
92 // if moving uphill (deltaHeight > 0) try fill up to the current height, otherwise deposit a fraction of the excess sediment
93 float amountToDeposit = (deltaHeight > 0) ? std::min(deltaHeight, sediment) : (sediment - sedimentCapacity) * depositSpeed;
94 sediment -= amountToDeposit;
95
96 // add the sediment to the four nodes of the current cell using bilinear interpolation
97 // deposition is not distributed over a radius (like erosion) so that it can fill small pits
98 (*mapData)[(size_t)dropletIndex] += amountToDeposit * (1 - cellOffsetX) * (1 - cellOffsetY);
99 (*mapData)[(size_t)dropletIndex + 1] += amountToDeposit * cellOffsetX * (1 - cellOffsetY);
100 (*mapData)[(size_t)dropletIndex + mapSize] += amountToDeposit * (1 - cellOffsetX) * cellOffsetY;
101 (*mapData)[(size_t)dropletIndex + mapSize + 1] += amountToDeposit * cellOffsetX * cellOffsetY;
102 }
103 else
104 {
105 // ERODE
106
107 // erode a fraction of the droplet's current carry capacity.
108 // clamp the erosion to the change in height so that it doesn't dig a hole in the terrain behind the droplet
109 float amountToErode = std::min((sedimentCapacity - sediment) * erodeSpeed, -deltaHeight);
110
111 // use erosion brush to erode from all nodes inside the droplet's erosion radius
112 for (int brushPointIndex = 0; brushPointIndex < (*erosionBrushIndices)[dropletIndex]->size(); brushPointIndex++)
113 {
114 int nodeIndex = (*(*erosionBrushIndices)[dropletIndex])[brushPointIndex];
115 float weighedErodeAmount = amountToErode * (*(*erosionBrushWeights)[dropletIndex])[brushPointIndex];
116 float deltaSediment = ((*mapData)[nodeIndex] < weighedErodeAmount) ? (*mapData)[nodeIndex] : weighedErodeAmount;
117 (*mapData)[nodeIndex] -= deltaSediment;
118 sediment += deltaSediment;
119 }
120 }
121
122 // update droplet's speed and water content
123 speed = sqrtf(speed * speed + deltaHeight * gravity);
124 if (isnan(speed))
125 speed = 0; // fix per alcuni NaN dovuti a speed * speed + deltaHeight * gravity negativo
126 water *= (1 - evaporateSpeed); // evaporate water
127 }
128 }

132void ErosionMaker::Gradient(std::vector<float>* mapData, int mapSize, float normalizedOffset, GradientType gradientType)
133{
134 float radius = ((float)mapSize / 2.0f);
135 for (size_t y = 0; y < mapSize; y++)
136 {
137 for (size_t x = 0; x < mapSize; x++)
138 {
139 int index = y * mapSize + x;
140 float gradient = 0.0f;
141 switch (gradientType) {
142 case GradientType::SQUARE:
143 gradient = std::max(std::abs((float)x - radius), std::abs((float)y - radius)) / (radius); // Chebyshev distance
144 break;
145
146 case GradientType::CIRCLE:
147 gradient = std::min(((x - radius) * (x - radius) + (y - radius) * (y - radius)) / (radius * radius), 1.0f); // Euclidean distance
148 break;
149
150 case GradientType::DIAMOND:
151 gradient = std::min((std::abs((float)x - radius) + std::abs((float)y - radius)) / (radius), 1.0f); // Manhattan distance
152 break;
153
154 case GradientType::STAR:
155 {
156 float g1 = std::min((std::abs((float)x - radius) + std::abs((float)y - radius)) / (radius), 1.0f); // Manhattan distance
157 float g2 = std::max(std::abs((float)x - radius), std::abs((float)y - radius)) / (radius); // Chebyshev distance
158 gradient = Lerp(g1, g2, 0.7f); // mix manhattan and chebyshev by desired value
159 }
160 break;
161
162 default:
163 gradient = std::min(((x - radius) * (x - radius) + (y - radius) * (y - radius)) / (radius * radius), 1.0f); // Euclidean distance
164 break;
165 }
166 gradient = 1 - gradient; // invert
167 (*mapData)[index] *= gradient; // multiply height by given linear gradient
168 }
169 }

172HeightAndGradient ErosionMaker::CalculateHeightAndGradient(std::vector<float>* mapData, int mapSize, float posX, float posY)
173{
174 int coordX = (int)posX;
175 int coordY = (int)posY;
176
177 // calculate droplet's offset inside the cell (0,0) = at NW node, (1,1) = at SE node
178 float x = posX - (float)coordX;
179 float y = posY - (float)coordY;
180
181 // calculate heights of the four nodes of the droplet's cell
182 int nodeIndexNW = coordY * mapSize + coordX;
183
184 float heightNW = (*mapData)[(size_t)nodeIndexNW];
185 float heightNE = (*mapData)[(size_t)nodeIndexNW + 1];
186 float heightSW = (*mapData)[(size_t)nodeIndexNW + mapSize];
187 float heightSE = (*mapData)[(size_t)nodeIndexNW + mapSize + 1];
188
189 // calculate droplet's direction of flow with bilinear interpolation of height difference along the edges
190 float gradientX = (heightNE - heightNW) * (1 - y) + (heightSE - heightSW) * y;
191 float gradientY = (heightSW - heightNW) * (1 - x) + (heightSE - heightNE) * x;
192
193 // calculate height with bilinear interpolation of the heights of the nodes of the cell
194 float height = heightNW * (1 - x) * (1 - y) + heightNE * x * (1 - y) + heightSW * (1 - x) * y + heightSE * x * y;
195
196 HeightAndGradient ret;
197 ret.height = height;
198 ret.gradientX = gradientX;
199 ret.gradientY = gradientY;
200 return ret;
201}

203void ErosionMaker::InitializeBrushIndices(int mapSize, int radius)
204{
205 erosionBrushIndices = new std::vector<std::vector<int>*>((size_t)mapSize * mapSize); // each cell stores its neighbors' indices
206 erosionBrushWeights = new std::vector<std::vector<float>*>((size_t)mapSize * mapSize); // each cell stores its neighbors' weight
207
208 std::vector<int> xOffsets((size_t)radius * radius * 4, 0);
209 std::vector<int> yOffsets((size_t)radius * radius * 4, 0);
210 std::vector<float> weights((size_t)radius * radius * 4, 0);
211 float weightSum = 0;
212 int addIndex = 0;
213
214 for (int i = 0; i < erosionBrushIndices->size(); i++) // va bene la prima coord
215 {
216 int centreX = i % mapSize; // x coordinate by cell
217 int centreY = i / mapSize; // y coodinate by cell
218
219 if (centreY <= radius || centreY >= mapSize - radius || centreX <= radius || centreX >= mapSize - radius) // loop only not too close to borders
220 {
221 weightSum = 0;
222 addIndex = 0;
223 for (int y = -radius; y <= radius; y++) // loop neighbors
224 {
225 for (int x = -radius; x <= radius; x++)
226 {
227 float sqrDst = x * x + y * y;
228 if (sqrDst < radius * radius) // take only those inside the radius of influence
229 {
230 int coordX = centreX + x;
231 int coordY = centreY + y;
232
233 if (coordX >= 0 && coordX < mapSize && coordY >= 0 && coordY < mapSize)
234 {
235 // add to brush index
236 float weight = 1 - sqrtf(sqrDst) / radius; // euclidean distance -> circle
237 weightSum += weight;
238 weights[addIndex] = weight;
239 xOffsets[addIndex] = x;
240 yOffsets[addIndex] = y;
241 addIndex++;
242 }
243 }
244 }
245 }
246 }
247
248 int numEntries = addIndex;
249 (*erosionBrushIndices)[i] = new std::vector<int>(numEntries);
250 (*erosionBrushWeights)[i] = new std::vector<float>(numEntries);
251
252 for (int j = 0; j < numEntries; j++)
253 {
254 (*(*erosionBrushIndices)[i])[j] = (yOffsets[j] + centreY) * mapSize + xOffsets[j] + centreX;
255 (*(*erosionBrushWeights)[i])[j] = weights[j] / weightSum;
256 }
257 }

326

328 // -1 0 1
329 // -2 0 2
