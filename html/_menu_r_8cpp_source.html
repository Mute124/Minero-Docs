11/*
12
13 The MenuR class creates static-size menu objects,
14 window-resized-not-responsive menus.
15
16 The menu object can contain one menu bar or several menu bars,
17 the menu can be automatically positioned on the left, center, or right side of the screen.
18 See mutators to modify the menu position.
19
20 The bars’ sizes are computed from the font size and length of the bar text longest;
21 the texts can be positioned on the left, center, or right side of the bars.
22 See mutators to modify each individual bar position, the text’s position in each individual bar,
23 and the each individual bar’s size.
24
25 The bars’ borders can be turned on and off.
26 The sizes of the borders are computed from the font size and length of the bar text longest bar;
27 modifying the bars’ sizes will also modify the borders’ sizes.
28
29 The button's shadows can be turned on and off.
30 The sizes of the shadows are computed from the font size and length of the nar text longest;
31 modifying the bars’ sizes will also modify the shadows’ sizes.
32
33 The default font is raylib font.
34
35 The menu bars are untilizing the ButtonR class
36
37

145//--------------------------------------------------------------------- Method buildMenu()
146/*----------------------------------------------------------
147
148 Builds the Menu,
149 computes size from the font size and length of the texts
150
151 -----------------------------------------------------------*/
152void MenuR::buildMenu()
153{
154 //---- Builds Bar
155 numBars =(unsigned)texts.size(); // the number of bars is computed from the size of the vector texts
156 //-- Bars' texts
157 if (isRayFont) // Default font
158 {
159 ButtonR bar; // empty bar, Default ButtonR
160 for (unsigned i = 0; i < numBars; i++) bars.push_back(bar);
161 for (unsigned i = 0; i < numBars; i++) bars[i].setText(texts[i]);
162 }
163 else // Loaded font
164 {
165 ButtonR bar("Empty", font, fontSize);
166 for (unsigned i = 0; i < numBars; i++) bars.push_back(bar);
167 for (unsigned i = 0; i < numBars; i++) bars[i].setText(texts[i]);
168 }
169
170 // Computes the bars width and menu width
171 // The bars’ sizes are computed from the font size and length of the longest text
172 float barsWidth = 0;
173 for (ButtonR &bar : bars) if (bar.rect.width > barsWidth) barsWidth = bar.rect.width + 50;
174 //-- Menu size
175 menuWidth = barsWidth;
176 menuHeight = numBars * bars[0].btnHeight + barSpacing * (numBars - 1);
177 //-- sets bars size
178 for (ButtonR &bar : bars) bar.setBtnSize(barsWidth, bar.rect.height);
179
180 //---- Menu position
181 int screenWidth = GetScreenWidth();
182 int screenHeight = GetScreenHeight();
183 float screenMargin = 50.0f;
184 try // Error handler
185 {
186 switch (menuPos) // Computes menu x and y position
187 {
188 case CENTER:
189 menuX = (float)(screenWidth / 2) - (menuWidth / 2);
190 menuY = (float)(screenHeight / 2) - (menuHeight / 2);
191 break;
192 case CENTER_TOP:
193 menuX = (float)(screenWidth / 2) - (menuWidth / 2);
194 menuY = screenMargin;
195 break;
196 case CENTER_BOTTOM:
197 menuX = (float)(screenWidth / 2) - (menuWidth / 2);
198 menuY = (float)(screenHeight - menuHeight - screenMargin);
199 break;
200 case LEFT:
201 menuX = screenMargin;
202 menuY = (float)(screenHeight / 2) - (menuHeight / 2);
203 break;
204 case LEFT_TOP:
205 menuX = screenMargin;
206 menuY = screenMargin;
207 break;
208 case LEFT_BOTTOM:
209 menuX = screenMargin;
210 menuY = (float)(screenHeight - menuHeight - screenMargin);
211 break;
212 case RIGHT:
213 menuX = (float)(screenWidth - menuWidth - 2 * screenMargin);
214 menuY = (float)(screenHeight / 2) - (menuHeight / 2);
215 break;
216 case RIGHT_TOP:
217 menuX = (float)(screenWidth - menuWidth - 2 * screenMargin);
218 menuY = screenMargin;
219 break;
220 case RIGHT_BOTTOM:
221 menuX = (float)(screenWidth - menuWidth - 2 * screenMargin);
222 menuY = (float)(screenHeight - menuHeight - screenMargin);
223 break;
224 case NONE:
225 // The menu was repositioned using setMenuPosition()
226 break;
227 default:
228 throw " -- Error --\n The entered menu position is not valid a valid entry.\n";
229 break;
230 }
231 } catch (char const *ERROR) { cout << "\n " << ERROR; }// Catches the exception if it occurs
232
233 //---- Bars positions
234 float tempBarY = menuY;
235 for (ButtonR &bar : bars)
236 {
237 bar.setBtnPosition(menuX, tempBarY);
238 tempBarY += bar.btnHeight + barSpacing;
239 }
240
241 //---- Texts position
242 for (ButtonR &bar : bars)
243 {
244 try // Error handler
245 {
246 switch (textsPos)
247 {
248 case TXT_CENTER:
249 // The ButtonR class centers the texts by default
250 break;
251 case TXT_LEFT:
252 bar.setTextPosition(bar.rect.x + 10, bar.textPos.y);
253 break;
254 case TXT_RIGHT:
255 bar.setTextPosition(bar.rect.x - 10 + (bar.rect.width - bar.textSize.x), bar.textPos.y);
256 break;
257 case TXT_NONE:
258 // The texts were repositioned using setTextPos()
259 break;
260 default:
261 throw " -- Error --\n The entered texts' position in menu is not a valid entry.\n";
262 break;
263 }
264 } catch (char const *ERROR) { cout << "\n " << ERROR; }// Catches the exception if it occurs
265 }
266
267}
