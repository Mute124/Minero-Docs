666void GenerateTrees(ErosionMaker* erosionMaker, std::vector<float>* mapData, Texture2D* treeTextures, std::vector<TreeBillboard>* trees, bool generateNew)
667{
668 Vector3 billPosition = { 0.0f, 0.0f, 0.0f };
669 Vector3 billNormal = { 0.0f, 0.0f, 0.0f };
670 float grassSlopeThreshold = 0.2; // different than in the terrain shader
671 float grassBlendAmount = 0.55;
672 float grassWeight;
673 Color billColor = WHITE;
674
675 for (size_t i = 0; i < TREE_COUNT; i++) // 8190 max billboards, more than that and they are not cached anymore
676 {
677 int px, py;
678 do
679 {
680 // try to generate a billboard
681 billPosition.x = randomRange(-16, 16);
682 billPosition.z = randomRange(-16, 16);
683 px = ((billPosition.x + 16.0f) / 32.0f) * (MAP_RESOLUTION - 1);
684 py = ((billPosition.z + 16.0f) / 32.0f) * (MAP_RESOLUTION - 1);
685 billNormal = erosionMaker->GetNormal(mapData, MAP_RESOLUTION, px, py);
686 billPosition.y = mapData->at(py * MAP_RESOLUTION + px) * 8 - 1.1f;
687
688 float slope = 1.0 - billNormal.y;
689 float grassBlendHeight = grassSlopeThreshold * (1.0 - grassBlendAmount);
690 grassWeight = 1.0 - std::min(std::max((slope - grassBlendHeight) / (grassSlopeThreshold - grassBlendHeight), 0.0f), 1.0f);
691 } while (billPosition.y < 0.32f || billPosition.y > 3.25f || grassWeight < 0.65f); // repeat until you find valid parameters (height and normal of chosen spot)
692
693 billColor.r = (billNormal.x + 1) * 127.5f; // terrain normal where tree is located, stored on color
694 billColor.g = (billNormal.y + 1) * 127.5f; // convert from range (-1, 1) to (0, 255)
695 billColor.b = (billNormal.z + 1) * 127.5f;
696
697 if (!generateNew)
698 {
699 (*trees)[i].position = billPosition;
700 (*trees)[i].color = billColor;
701 }
702 else
703 {
704 int textureChoice = (int)randomRange(0, TREE_TEXTURE_COUNT);
705 trees->push_back({ treeTextures[textureChoice], billPosition, randomRange(0.6f, 1.4f) * 0.3f, billColor });
706 }
707 }
708}
