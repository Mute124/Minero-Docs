149 template<typename genIUType>
150 GLM_FUNC_QUALIFIER genIUType highestBitValue(genIUType Value)
151 {
152 genIUType tmp = Value;
153 genIUType result = genIUType(0);
154 while(tmp)
155 {
156 result = (tmp & (~tmp + 1)); // grab lowest bit
157 tmp &= ~result; // clear lowest bit
158 }
159 return result;
160 }

162 template<typename genType>
163 GLM_FUNC_QUALIFIER genType ceilPowerOfTwo_loop(genType value)
164 {
165 return glm::isPowerOfTwo(value) ? value : highestBitValue(value) << 1;
