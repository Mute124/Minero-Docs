85// Apply chosen cursor mode to a focused window
86//
87static void updateCursorMode(_GLFWwindow* window)
88{
89 if (window->cursorMode == GLFW_CURSOR_DISABLED)
90 {
91 _glfw.ns.disabledCursorWindow = window;
92 _glfwGetCursorPosCocoa(window,
93 &_glfw.ns.restoreCursorPosX,
94 &_glfw.ns.restoreCursorPosY);
95 _glfwCenterCursorInContentArea(window);
96 CGAssociateMouseAndMouseCursorPosition(false);
97 }
98 else if (_glfw.ns.disabledCursorWindow == window)
99 {
100 _glfw.ns.disabledCursorWindow = NULL;
101 _glfwSetCursorPosCocoa(window,
102 _glfw.ns.restoreCursorPosX,
103 _glfw.ns.restoreCursorPosY);
104 // NOTE: The matching CGAssociateMouseAndMouseCursorPosition call is
105 // made in _glfwSetCursorPosCocoa as part of a workaround
106 }
107
108 if (cursorInContentArea(window))
109 updateCursorImage(window);
110}

139// Translates macOS key modifiers into GLFW ones
140//
141static int translateFlags(NSUInteger flags)
142{
143 int mods = 0;
144
145 if (flags & NSEventModifierFlagShift)
146 mods |= GLFW_MOD_SHIFT;
147 if (flags & NSEventModifierFlagControl)
148 mods |= GLFW_MOD_CONTROL;
149 if (flags & NSEventModifierFlagOption)
150 mods |= GLFW_MOD_ALT;
151 if (flags & NSEventModifierFlagCommand)
152 mods |= GLFW_MOD_SUPER;
153 if (flags & NSEventModifierFlagCapsLock)
154 mods |= GLFW_MOD_CAPS_LOCK;
155
156 return mods;
157}

159// Translates a macOS keycode to a GLFW keycode
160//
161static int translateKey(unsigned int key)
162{
163 if (key >= sizeof(_glfw.ns.keycodes) / sizeof(_glfw.ns.keycodes[0]))
164 return GLFW_KEY_UNKNOWN;
165
166 return _glfw.ns.keycodes[key];
167}

169// Translate a GLFW keycode to a Cocoa modifier flag
170//
171static NSUInteger translateKeyToModifierFlag(int key)
172{
173 switch (key)
174 {
175 case GLFW_KEY_LEFT_SHIFT:
176 case GLFW_KEY_RIGHT_SHIFT:
177 return NSEventModifierFlagShift;
178 case GLFW_KEY_LEFT_CONTROL:
179 case GLFW_KEY_RIGHT_CONTROL:
180 return NSEventModifierFlagControl;
181 case GLFW_KEY_LEFT_ALT:
182 case GLFW_KEY_RIGHT_ALT:
183 return NSEventModifierFlagOption;
184 case GLFW_KEY_LEFT_SUPER:
185 case GLFW_KEY_RIGHT_SUPER:
186 return NSEventModifierFlagCommand;
187 case GLFW_KEY_CAPS_LOCK:
188 return NSEventModifierFlagCapsLock;
189 }
190
191 return 0;
192}

618- (BOOL)performDragOperation:(id <NSDraggingInfo>)sender
619{
620 const NSRect contentRect = [window->ns.view frame];
621 // NOTE: The returned location uses base 0,1 not 0,0
622 const NSPoint pos = [sender draggingLocation];
623 _glfwInputCursorPos(window, pos.x, contentRect.size.height - pos.y);
624
625 NSPasteboard* pasteboard = [sender draggingPasteboard];
626 NSDictionary* options = @{NSPasteboardURLReadingFileURLsOnlyKey:@YES};
627 NSArray* urls = [pasteboard readObjectsForClasses:@[[NSURL class]]
628 options:options];
629 const NSUInteger count = [urls count];
630 if (count)
631 {
632 char** paths = _glfw_calloc(count, sizeof(char*));
633
634 for (NSUInteger i = 0; i < count; i++)
635 paths[i] = _glfw_strdup([urls[i] fileSystemRepresentation]);
636
637 _glfwInputDrop(window, (int) count, (const char**) paths);
638
639 for (NSUInteger i = 0; i < count; i++)
640 _glfw_free(paths[i]);
641 _glfw_free(paths);
642 }
643
644 return YES;
645}

704- (void)insertText:(id)string replacementRange:(NSRange)replacementRange
705{
706 NSString* characters;
707 NSEvent* event = [NSApp currentEvent];
708 const int mods = translateFlags([event modifierFlags]);
709 const int plain = !(mods & GLFW_MOD_SUPER);
710
711 if ([string isKindOfClass:[NSAttributedString class]])
712 characters = [string string];
713 else
714 characters = (NSString*) string;
715
716 NSRange range = NSMakeRange(0, [characters length]);
717 while (range.length)
718 {
719 uint32_t codepoint = 0;
720
721 if ([characters getBytes:&codepoint
722 maxLength:sizeof(codepoint)
723 usedLength:NULL
724 encoding:NSUTF32StringEncoding
725 options:0
726 range:range
727 remainingRange:&range])
728 {
729 if (codepoint >= 0xf700 && codepoint <= 0xf7ff)
730 continue;
731
732 _glfwInputChar(window, codepoint, mods, plain);
733 }
734 }

767// Create the Cocoa window
768//
769static GLFWbool createNativeWindow(_GLFWwindow* window,
770 const _GLFWwndconfig* wndconfig,
771 const _GLFWfbconfig* fbconfig)
772{
773 window->ns.delegate = [[GLFWWindowDelegate alloc] initWithGlfwWindow:window];
774 if (window->ns.delegate == nil)
775 {
776 _glfwInputError(GLFW_PLATFORM_ERROR,
777 "Cocoa: Failed to create window delegate");
778 return GLFW_FALSE;
779 }
780
781 NSRect contentRect;
782
783 if (window->monitor)
784 {
785 GLFWvidmode mode;
786 int xpos, ypos;
787
788 _glfwGetVideoModeCocoa(window->monitor, &mode);
789 _glfwGetMonitorPosCocoa(window->monitor, &xpos, &ypos);
790
791 contentRect = NSMakeRect(xpos, ypos, mode.width, mode.height);
792 }
793 else
794 {
795 if (wndconfig->xpos == GLFW_ANY_POSITION ||
796 wndconfig->ypos == GLFW_ANY_POSITION)
797 {
798 contentRect = NSMakeRect(0, 0, wndconfig->width, wndconfig->height);
799 }
800 else
801 {
802 const int xpos = wndconfig->xpos;
803 const int ypos = _glfwTransformYCocoa(wndconfig->ypos + wndconfig->height - 1);
804 contentRect = NSMakeRect(xpos, ypos, wndconfig->width, wndconfig->height);
805 }
806 }
807
808 NSUInteger styleMask = NSWindowStyleMaskMiniaturizable;
809
810 if (window->monitor || !window->decorated)
811 styleMask |= NSWindowStyleMaskBorderless;
812 else
813 {
814 styleMask |= (NSWindowStyleMaskTitled | NSWindowStyleMaskClosable);
815
816 if (window->resizable)
817 styleMask |= NSWindowStyleMaskResizable;
818 }
819
820 window->ns.object = [[GLFWWindow alloc]
821 initWithContentRect:contentRect
822 styleMask:styleMask
823 backing:NSBackingStoreBuffered
824 defer:NO];
825
826 if (window->ns.object == nil)
827 {
828 _glfwInputError(GLFW_PLATFORM_ERROR, "Cocoa: Failed to create window");
829 return GLFW_FALSE;
830 }
831
832 if (window->monitor)
833 [window->ns.object setLevel:NSMainMenuWindowLevel + 1];
834 else
835 {
836 if (wndconfig->xpos == GLFW_ANY_POSITION ||
837 wndconfig->ypos == GLFW_ANY_POSITION)
838 {
