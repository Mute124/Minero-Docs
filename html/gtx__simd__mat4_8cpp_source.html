219int test_compute_gtx()
220{
221 std::vector<glm::vec4> Output(1000000);
222
223 std::clock_t TimeStart = clock();
224
225 for(std::size_t k = 0; k < Output.size(); ++k)
226 {
227 float i = float(k) / 1000.f + 0.001f;
228 glm::vec3 A = glm::normalize(glm::vec3(i));
229 glm::vec3 B = glm::cross(A, glm::normalize(glm::vec3(1, 1, 2)));
230 glm::mat4 C = glm::rotate(glm::mat4(1.0f), i, B);
231 glm::mat4 D = glm::scale(C, glm::vec3(0.8f, 1.0f, 1.2f));
232 glm::mat4 E = glm::translate(D, glm::vec3(1.4f, 1.2f, 1.1f));
233 glm::mat4 F = glm::perspective(i, 1.5f, 0.1f, 1000.f);
234 glm::mat4 G = glm::inverse(F * E);
235 glm::vec3 H = glm::unProject(glm::vec3(i), G, F, E[3]);
236 glm::vec3 I = glm::any(glm::isnan(glm::project(H, G, F, E[3]))) ? glm::vec3(2) : glm::vec3(1);
237 glm::mat4 J = glm::lookAt(glm::normalize(glm::max(B, glm::vec3(0.001f))), H, I);
238 glm::mat4 K = glm::transpose(J);
239 glm::quat L = glm::normalize(glm::quat_cast(K));
240 glm::vec4 M = L * glm::smoothstep(K[3], J[3], glm::vec4(i));
241 glm::mat4 N = glm::mat4(glm::normalize(glm::max(M, glm::vec4(0.001f))), K[3], J[3], glm::vec4(i));
242 glm::mat4 O = N * glm::inverse(N);
243 glm::vec4 P = O * glm::reflect(N[3], glm::vec4(A, 1.0f));
244 glm::vec4 Q = glm::vec4(glm::dot(M, P));
245 glm::vec4 R = glm::quat(Q.w, glm::vec3(Q)) * P;
246 Output[k] = R;
247 }
248
249 std::clock_t TimeEnd = clock();
250 printf("test_compute_gtx: %ld\n", TimeEnd - TimeStart);
251
252 return 0;
253}
