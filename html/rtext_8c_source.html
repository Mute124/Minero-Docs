299// Load Font from file into GPU memory (VRAM)
300Font LoadFont(const char *fileName)
301{
302 // Default values for ttf font generation
303#ifndef FONT_TTF_DEFAULT_SIZE
304 #define FONT_TTF_DEFAULT_SIZE 32 // TTF font generation default char size (char-height)
305#endif
306#ifndef FONT_TTF_DEFAULT_NUMCHARS
307 #define FONT_TTF_DEFAULT_NUMCHARS 95 // TTF font generation default charset: 95 glyphs (ASCII 32..126)
308#endif
309#ifndef FONT_TTF_DEFAULT_FIRST_CHAR
310 #define FONT_TTF_DEFAULT_FIRST_CHAR 32 // TTF font generation default first char for image sprite font (32-Space)
311#endif
312#ifndef FONT_TTF_DEFAULT_CHARS_PADDING
313 #define FONT_TTF_DEFAULT_CHARS_PADDING 4 // TTF font generation default chars padding
314#endif
315
316 Font font = { 0 };
317
318#if defined(SUPPORT_FILEFORMAT_TTF)
319 if (IsFileExtension(fileName, ".ttf") || IsFileExtension(fileName, ".otf")) font = LoadFontEx(fileName, FONT_TTF_DEFAULT_SIZE, NULL, FONT_TTF_DEFAULT_NUMCHARS);
320 else
321#endif
322#if defined(SUPPORT_FILEFORMAT_FNT)
323 if (IsFileExtension(fileName, ".fnt")) font = LoadBMFont(fileName);
324 else
325#endif
326 {
327 Image image = LoadImage(fileName);
328 if (image.data != NULL) font = LoadFontFromImage(image, MAGENTA, FONT_TTF_DEFAULT_FIRST_CHAR);
329 UnloadImage(image);
330 }
331
332 if (font.texture.id == 0)
333 {
334 TRACELOG(LOG_WARNING, "FONT: [%s] Failed to load font texture -> Using default font", fileName);
335 font = GetFontDefault();
336 }
337 else
338 {
339 SetTextureFilter(font.texture, TEXTURE_FILTER_POINT); // By default, we set point filter (the best performance)
340 TRACELOG(LOG_INFO, "FONT: Data loaded successfully (%i pixel size | %i glyphs)", FONT_TTF_DEFAULT_SIZE, FONT_TTF_DEFAULT_NUMCHARS);
341 }
342
343 return font;
344}

369// Load an Image font file (XNA style)
370Font LoadFontFromImage(Image image, Color key, int firstChar)
371{
372#ifndef MAX_GLYPHS_FROM_IMAGE
373 #define MAX_GLYPHS_FROM_IMAGE 256 // Maximum number of glyphs supported on image scan
374#endif
375
376 #define COLOR_EQUAL(col1, col2) ((col1.r == col2.r) && (col1.g == col2.g) && (col1.b == col2.b) && (col1.a == col2.a))
377
378 Font font = GetFontDefault();
379
380 int charSpacing = 0;
381 int lineSpacing = 0;
382
383 int x = 0;
384 int y = 0;
385
386 // We allocate a temporal arrays for chars data measures,
387 // once we get the actual number of chars, we copy data to a sized arrays
388 int tempCharValues[MAX_GLYPHS_FROM_IMAGE] = { 0 };
389 Rectangle tempCharRecs[MAX_GLYPHS_FROM_IMAGE] = { 0 };
390
391 Color *pixels = LoadImageColors(image);
392
393 // Parse image data to get charSpacing and lineSpacing
394 for (y = 0; y < image.height; y++)
395 {
396 for (x = 0; x < image.width; x++)
397 {
398 if (!COLOR_EQUAL(pixels[y*image.width + x], key)) break;
399 }
400
401 if (!COLOR_EQUAL(pixels[y*image.width + x], key)) break;
402 }
403
404 if ((x == 0) || (y == 0)) return font;
405
406 charSpacing = x;
407 lineSpacing = y;
408
409 int charHeight = 0;
410 int j = 0;
411
412 while (!COLOR_EQUAL(pixels[(lineSpacing + j)*image.width + charSpacing], key)) j++;
413
414 charHeight = j;
415
416 // Check array values to get characters: value, x, y, w, h
417 int index = 0;
418 int lineToRead = 0;
419 int xPosToRead = charSpacing;
420
421 // Parse image data to get rectangle sizes
422 while ((lineSpacing + lineToRead*(charHeight + lineSpacing)) < image.height)
423 {
424 while ((xPosToRead < image.width) &&
425 !COLOR_EQUAL((pixels[(lineSpacing + (charHeight+lineSpacing)*lineToRead)*image.width + xPosToRead]), key))
426 {
427 tempCharValues[index] = firstChar + index;
428
429 tempCharRecs[index].x = (float)xPosToRead;
430 tempCharRecs[index].y = (float)(lineSpacing + lineToRead*(charHeight + lineSpacing));
431 tempCharRecs[index].height = (float)charHeight;
432
433 int charWidth = 0;
434
435 while (!COLOR_EQUAL(pixels[(lineSpacing + (charHeight+lineSpacing)*lineToRead)*image.width + xPosToRead + charWidth], key)) charWidth++;
436
437 tempCharRecs[index].width = (float)charWidth;
438
439 index++;
440
441 xPosToRead += (charWidth + charSpacing);
442 }
443
444 lineToRead++;
445 xPosToRead = charSpacing;
446 }
447
448 // NOTE: We need to remove key color borders from image to avoid weird
449 // artifacts on texture scaling when using TEXTURE_FILTER_BILINEAR or TEXTURE_FILTER_TRILINEAR
450 for (int i = 0; i < image.height*image.width; i++) if (COLOR_EQUAL(pixels[i], key)) pixels[i] = BLANK;
451
452 // Create a new image with the processed color data (key color replaced by BLANK)
453 Image fontClear = {
454 .data = pixels,
455 .width = image.width,
456 .height = image.height,
457 .mipmaps = 1,
458 .format = PIXELFORMAT_UNCOMPRESSED_R8G8B8A8
459 };
460
461 // Set font with all data parsed from image
462 font.texture = LoadTextureFromImage(fontClear); // Convert processed image to OpenGL texture
463 font.glyphCount = index;
464 font.glyphPadding = 0;
465
466 // We got tempCharValues and tempCharsRecs populated with chars data
467 // Now we move temp data to sized charValues and charRecs arrays
468 font.glyphs = (GlyphInfo *)RL_MALLOC(font.glyphCount*sizeof(GlyphInfo));
469 font.recs = (Rectangle *)RL_MALLOC(font.glyphCount*sizeof(Rectangle));
470
471 for (int i = 0; i < font.glyphCount; i++)
472 {
473 font.glyphs[i].value = tempCharValues[i];
474
475 // Get character rectangle in the font atlas texture
476 font.recs[i] = tempCharRecs[i];
477
478 // NOTE: On image based fonts (XNA style), character offsets and xAdvance are not required (set to 0)
479 font.glyphs[i].offsetX = 0;
480 font.glyphs[i].offsetY = 0;
481 font.glyphs[i].advanceX = 0;
482
483 // Fill character image data from fontClear data
484 font.glyphs[i].image = ImageFromImage(fontClear, tempCharRecs[i]);
485 }
486
487 UnloadImage(fontClear); // Unload processed image once converted to texture
488
489 font.baseSize = (int)font.recs[0].height;
490
491 return font;
492}

495Font LoadFontFromMemory(const char *fileType, const unsigned char *fileData, int dataSize, int fontSize, int *fontChars, int glyphCount)
496{
497 Font font = { 0 };
498
499 char fileExtLower[16] = { 0 };
500 strcpy(fileExtLower, TextToLower(fileType));
501
502#if defined(SUPPORT_FILEFORMAT_TTF)

504 TextIsEqual(fileExtLower, ".otf"))
505 {
506 font.baseSize = fontSize;
507 font.glyphCount = (glyphCount > 0)? glyphCount : 95;
508 font.glyphPadding = 0;
509 font.glyphs = LoadFontData(fileData, dataSize, font.baseSize, fontChars, font.glyphCount, FONT_DEFAULT);

551// Load font data for further use
552// NOTE: Requires TTF font memory data and can generate SDF data
553GlyphInfo *LoadFontData(const unsigned char *fileData, int dataSize, int fontSize, int *fontChars, int glyphCount, int type)
554{
555 // NOTE: Using some SDF generation default values,
556 // trades off precision with ability to handle *smaller* sizes
557#ifndef FONT_SDF_CHAR_PADDING
558 #define FONT_SDF_CHAR_PADDING 4 // SDF font generation char padding
559#endif
560#ifndef FONT_SDF_ON_EDGE_VALUE
561 #define FONT_SDF_ON_EDGE_VALUE 128 // SDF font generation on edge value
562#endif
563#ifndef FONT_SDF_PIXEL_DIST_SCALE
564 #define FONT_SDF_PIXEL_DIST_SCALE 64.0f // SDF font generation pixel distance scale
565#endif
566#ifndef FONT_BITMAP_ALPHA_THRESHOLD
567 #define FONT_BITMAP_ALPHA_THRESHOLD 80 // Bitmap (B&W) font generation alpha threshold
568#endif
569
570 GlyphInfo *chars = NULL;
571
572#if defined(SUPPORT_FILEFORMAT_TTF)
573 // Load font data (including pixel data) from TTF memory file
574 // NOTE: Loaded information should be enough to generate font image atlas, using any packaging method
575 if (fileData != NULL)
576 {
577 bool genFontChars = false;
578 stbtt_fontinfo fontInfo = { 0 };
579
580 if (stbtt_InitFont(&fontInfo, (unsigned char *)fileData, 0)) // Initialize font for data reading
581 {
582 // Calculate font scale factor
583 float scaleFactor = stbtt_ScaleForPixelHeight(&fontInfo, (float)fontSize);
584
585 // Calculate font basic metrics
586 // NOTE: ascent is equivalent to font baseline
587 int ascent, descent, lineGap;
588 stbtt_GetFontVMetrics(&fontInfo, &ascent, &descent, &lineGap);
589
590 // In case no chars count provided, default to 95
591 glyphCount = (glyphCount > 0)? glyphCount : 95;
592
593 // Fill fontChars in case not provided externally
594 // NOTE: By default we fill glyphCount consecutively, starting at 32 (Space)
595
596 if (fontChars == NULL)
597 {
598 fontChars = (int *)RL_MALLOC(glyphCount*sizeof(int));
599 for (int i = 0; i < glyphCount; i++) fontChars[i] = i + 32;
600 genFontChars = true;
601 }
602
603 chars = (GlyphInfo *)RL_MALLOC(glyphCount*sizeof(GlyphInfo));
604
605 // NOTE: Using simple packaging, one char after another
606 for (int i = 0; i < glyphCount; i++)
607 {
608 int chw = 0, chh = 0; // Character width and height (on generation)
609 int ch = fontChars[i]; // Character value to get info for
610 chars[i].value = ch;
611
612 // Render a unicode codepoint to a bitmap
613 // stbtt_GetCodepointBitmap() -- allocates and returns a bitmap
614 // stbtt_GetCodepointBitmapBox() -- how big the bitmap must be
615 // stbtt_MakeCodepointBitmap() -- renders into bitmap you provide
616
617 if (type != FONT_SDF) chars[i].image.data = stbtt_GetCodepointBitmap(&fontInfo, scaleFactor, scaleFactor, ch, &chw, &chh, &chars[i].offsetX, &chars[i].offsetY);
618 else if (ch != 32) chars[i].image.data = stbtt_GetCodepointSDF(&fontInfo, scaleFactor, ch, FONT_SDF_CHAR_PADDING, FONT_SDF_ON_EDGE_VALUE, FONT_SDF_PIXEL_DIST_SCALE, &chw, &chh, &chars[i].offsetX, &chars[i].offsetY);
619 else chars[i].image.data = NULL;
620
621 stbtt_GetCodepointHMetrics(&fontInfo, ch, &chars[i].advanceX, NULL);
622 chars[i].advanceX = (int)((float)chars[i].advanceX*scaleFactor);
623
624 // Load characters images
625 chars[i].image.width = chw;
626 chars[i].image.height = chh;
627 chars[i].image.mipmaps = 1;
628 chars[i].image.format = PIXELFORMAT_UNCOMPRESSED_GRAYSCALE;
629
630 chars[i].offsetY += (int)((float)ascent*scaleFactor);
631
632 // NOTE: We create an empty image for space character, it could be further required for atlas packing
633 if (ch == 32)
634 {
635 Image imSpace = {
636 .data = RL_CALLOC(chars[i].advanceX*fontSize, 2),
637 .width = chars[i].advanceX,
638 .height = fontSize,
639 .mipmaps = 1,
640 .format = PIXELFORMAT_UNCOMPRESSED_GRAYSCALE
641 };
642
643 chars[i].image = imSpace;
644 }
645
646 if (type == FONT_BITMAP)
647 {
648 // Aliased bitmap (black & white) font generation, avoiding anti-aliasing
649 // NOTE: For optimum results, bitmap font should be generated at base pixel size
650 for (int p = 0; p < chw*chh; p++)
651 {
652 if (((unsigned char *)chars[i].image.data)[p] < FONT_BITMAP_ALPHA_THRESHOLD) ((unsigned char *)chars[i].image.data)[p] = 0;
653 else ((unsigned char *)chars[i].image.data)[p] = 255;
654 }
655 }
656
657 // Get bounding box for character (maybe offset to account for chars that dip above or below the line)
658 /*
659 int chX1, chY1, chX2, chY2;
660 stbtt_GetCodepointBitmapBox(&fontInfo, ch, scaleFactor, scaleFactor, &chX1, &chY1, &chX2, &chY2);
661
662 TRACELOGD("FONT: Character box measures: %i, %i, %i, %i", chX1, chY1, chX2 - chX1, chY2 - chY1);
663 TRACELOGD("FONT: Character offsetY: %i", (int)((float)ascent*scaleFactor) + chY1);
664 */
665 }
666 }
667 else TRACELOG(LOG_WARNING, "FONT: Failed to process TTF font data");
668
669 if (genFontChars) RL_FREE(fontChars);
670 }
671#endif
672
673 return chars;
674}

676// Generate image font atlas using chars info
677// NOTE: Packing method: 0-Default, 1-Skyline
678#if defined(SUPPORT_FILEFORMAT_TTF)
679Image GenImageFontAtlas(const GlyphInfo *chars, Rectangle **charRecs, int glyphCount, int fontSize, int padding, int packMethod)
680{
681 Image atlas = { 0 };
682
683 if (chars == NULL)
684 {
685 TRACELOG(LOG_WARNING, "FONT: Provided chars info not valid, returning empty image atlas");
686 return atlas;
687 }
688
689 *charRecs = NULL;
690
691 // In case no chars count provided we suppose default of 95
692 glyphCount = (glyphCount > 0)? glyphCount : 95;
693
694 // NOTE: Rectangles memory is loaded here!
695 Rectangle *recs = (Rectangle *)RL_MALLOC(glyphCount*sizeof(Rectangle));
696
697 // Calculate image size based on required pixel area
698 // NOTE 1: Image is forced to be squared and POT... very conservative!
699 // NOTE 2: SDF font characters already contain an internal padding,
700 // so image size would result bigger than default font type
701 float requiredArea = 0;
702 for (int i = 0; i < glyphCount; i++) requiredArea += ((chars[i].image.width + 2*padding)*(fontSize + 2*padding));
703 float guessSize = sqrtf(requiredArea)*1.4f;
704 int imageSize = (int)powf(2, ceilf(logf((float)guessSize)/logf(2))); // Calculate next POT
705
706 atlas.width = imageSize; // Atlas bitmap width
707 atlas.height = imageSize; // Atlas bitmap height
708 atlas.data = (unsigned char *)RL_CALLOC(1, atlas.width*atlas.height); // Create a bitmap to store characters (8 bpp)
709 atlas.format = PIXELFORMAT_UNCOMPRESSED_GRAYSCALE;
710 atlas.mipmaps = 1;
711
712 // DEBUG: We can see padding in the generated image setting a gray background...
713 //for (int i = 0; i < atlas.width*atlas.height; i++) ((unsigned char *)atlas.data)[i] = 100;
714
715 if (packMethod == 0) // Use basic packing algorithm
716 {
717 int offsetX = padding;
718 int offsetY = padding;
719
720 // NOTE: Using simple packaging, one char after another
721 for (int i = 0; i < glyphCount; i++)
722 {
723 // Copy pixel data from fc.data to atlas
724 for (int y = 0; y < chars[i].image.height; y++)
725 {
726 for (int x = 0; x < chars[i].image.width; x++)
727 {
728 ((unsigned char *)atlas.data)[(offsetY + y)*atlas.width + (offsetX + x)] = ((unsigned char *)chars[i].image.data)[y*chars[i].image.width + x];
729 }
730 }
731
732 // Fill chars rectangles in atlas info
733 recs[i].x = (float)offsetX;
734 recs[i].y = (float)offsetY;
735 recs[i].width = (float)chars[i].image.width;
736 recs[i].height = (float)chars[i].image.height;
737
738 // Move atlas position X for next character drawing
739 offsetX += (chars[i].image.width + 2*padding);
740
741 if (offsetX >= (atlas.width - chars[i].image.width - 2*padding))
742 {
743 offsetX = padding;
744
745 // NOTE: Be careful on offsetY for SDF fonts, by default SDF
746 // use an internal padding of 4 pixels, it means char rectangle
747 // height is bigger than fontSize, it could be up to (fontSize + 8)
748 offsetY += (fontSize + 2*padding);
749
750 if (offsetY > (atlas.height - fontSize - padding))
751 {
752 for(int j = i + 1; j < glyphCount; j++)
753 {
754 TRACELOG(LOG_WARNING, "FONT: Failed to package character (%i)", j);
755 // make sure remaining recs contain valid data
756 recs[j].x = 0;
757 recs[j].y = 0;
758 recs[j].width = 0;
759 recs[j].height = 0;
760 }
761 break;
762 }
763 }
764 }
765 }
766 else if (packMethod == 1) // Use Skyline rect packing algorithm (stb_pack_rect)
767 {
768 stbrp_context *context = (stbrp_context *)RL_MALLOC(sizeof(*context));
769 stbrp_node *nodes = (stbrp_node *)RL_MALLOC(glyphCount*sizeof(*nodes));
770
771 stbrp_init_target(context, atlas.width, atlas.height, nodes, glyphCount);
772 stbrp_rect *rects = (stbrp_rect *)RL_MALLOC(glyphCount*sizeof(stbrp_rect));
773
774 // Fill rectangles for packaging
775 for (int i = 0; i < glyphCount; i++)
776 {
777 rects[i].id = i;
778 rects[i].w = chars[i].image.width + 2*padding;
779 rects[i].h = chars[i].image.height + 2*padding;
780 }
781
782 // Package rectangles into atlas
783 stbrp_pack_rects(context, rects, glyphCount);
784
785 for (int i = 0; i < glyphCount; i++)
786 {
787 // It returns char rectangles in atlas
788 recs[i].x = rects[i].x + (float)padding;
789 recs[i].y = rects[i].y + (float)padding;
790 recs[i].width = (float)chars[i].image.width;
791 recs[i].height = (float)chars[i].image.height;
792
793 if (rects[i].was_packed)
794 {
795 // Copy pixel data from fc.data to atlas
796 for (int y = 0; y < chars[i].image.height; y++)
797 {
798 for (int x = 0; x < chars[i].image.width; x++)
799 {
800 ((unsigned char *)atlas.data)[(rects[i].y + padding + y)*atlas.width + (rects[i].x + padding + x)] = ((unsigned char *)chars[i].image.data)[y*chars[i].image.width + x];
801 }
802 }
803 }
804 else TRACELOG(LOG_WARNING, "FONT: Failed to package character (%i)", i);
805 }
806
807 RL_FREE(rects);
808 RL_FREE(nodes);
809 RL_FREE(context);
810 }
811
812 // Convert image data from GRAYSCALE to GRAY_ALPHA
813 unsigned char *dataGrayAlpha = (unsigned char *)RL_MALLOC(atlas.width*atlas.height*sizeof(unsigned char)*2); // Two channels
814
815 for (int i = 0, k = 0; i < atlas.width*atlas.height; i++, k += 2)
816 {
817 dataGrayAlpha[k] = 255;
818 dataGrayAlpha[k + 1] = ((unsigned char *)atlas.data)[i];
819 }
820
821 RL_FREE(atlas.data);
822 atlas.data = dataGrayAlpha;
823 atlas.format = PIXELFORMAT_UNCOMPRESSED_GRAY_ALPHA;
824
825 *charRecs = recs;
826
827 return atlas;
828}

1407// Replace text string
1408// REQUIRES: strlen(), strstr(), strncpy(), strcpy()
1409// WARNING: Allocated memory must be manually freed
1410char *TextReplace(char *text, const char *replace, const char *by)
1411{
1412 // Sanity checks and initialization
1413 if (!text || !replace || !by) return NULL;
1414
1415 char *result = NULL;
1416
1417 char *insertPoint = NULL; // Next insert point
1418 char *temp = NULL; // Temp pointer
1419 int replaceLen = 0; // Replace string length of (the string to remove)
1420 int byLen = 0; // Replacement length (the string to replace by)
1421 int lastReplacePos = 0; // Distance between replace and end of last replace
1422 int count = 0; // Number of replacements
1423
1424 replaceLen = TextLength(replace);
1425 if (replaceLen == 0) return NULL; // Empty replace causes infinite loop during count
1426
1427 byLen = TextLength(by);
1428
1429 // Count the number of replacements needed
1430 insertPoint = text;
1431 for (count = 0; (temp = strstr(insertPoint, replace)); count++) insertPoint = temp + replaceLen;
1432
1433 // Allocate returning string and point temp to it
1434 temp = result = (char *)RL_MALLOC(TextLength(text) + (byLen - replaceLen)*count + 1);
1435
1436 if (!result) return NULL; // Memory could not be allocated
1437
1438 // First time through the loop, all the variable are set correctly from here on,
1439 // - 'temp' points to the end of the result string
1440 // - 'insertPoint' points to the next occurrence of replace in text
1441 // - 'text' points to the remainder of text after "end of replace"
1442 while (count--)
1443 {
1444 insertPoint = strstr(text, replace);
1445 lastReplacePos = (int)(insertPoint - text);
1446 temp = strncpy(temp, text, lastReplacePos) + lastReplacePos;
1447 temp = strcpy(temp, by) + byLen;
1448 text += lastReplacePos + replaceLen; // Move to next "end of replace"
1449 }
1450
1451 // Copy remaind text part after replacement to result (pointed by moving temp)
1452 strcpy(temp, text);
1453
1454 return result;
1455}

1774// Get next codepoint in a UTF-8 encoded text, scanning until '\0' is found
1775// When an invalid UTF-8 byte is encountered we exit as soon as possible and a '?'(0x3f) codepoint is returned
1776// Total number of bytes processed are returned as a parameter
1777// NOTE: The standard says U+FFFD should be returned in case of errors
1778// but that character is not supported by the default font in raylib
1779int GetCodepoint(const char *text, int *codepointSize)
1780{
1781/*
1782 UTF-8 specs from https://www.ietf.org/rfc/rfc3629.txt
1783
1784 Char. number range | UTF-8 octet sequence
1785 (hexadecimal) | (binary)
1786 --------------------+---------------------------------------------
1787 0000 0000-0000 007F | 0xxxxxxx
1788 0000 0080-0000 07FF | 110xxxxx 10xxxxxx
1789 0000 0800-0000 FFFF | 1110xxxx 10xxxxxx 10xxxxxx
1790 0001 0000-0010 FFFF | 11110xxx 10xxxxxx 10xxxxxx 10xxxxxx
1791*/
1792 // NOTE: on decode errors we return as soon as possible
1793
1794 int codepoint = 0x3f; // Codepoint (defaults to '?')
1795 int octet = (unsigned char)(text[0]); // The first UTF8 octet
1796 *codepointSize = 1;
1797
1798 if (octet <= 0x7f)
1799 {
1800 // Only one octet (ASCII range x00-7F)
1801 codepoint = text[0];
1802 }
1803 else if ((octet & 0xe0) == 0xc0)
1804 {
1805 // Two octets
1806
1807 // [0]xC2-DF [1]UTF8-tail(x80-BF)
1808 unsigned char octet1 = text[1];
1809
1810 if ((octet1 == '\0') || ((octet1 >> 6) != 2)) { *codepointSize = 2; return codepoint; } // Unexpected sequence
1811
1812 if ((octet >= 0xc2) && (octet <= 0xdf))
1813 {
1814 codepoint = ((octet & 0x1f) << 6) | (octet1 & 0x3f);
1815 *codepointSize = 2;
1816 }
1817 }
1818 else if ((octet & 0xf0) == 0xe0)
1819 {
1820 // Three octets
1821 unsigned char octet1 = text[1];
1822 unsigned char octet2 = '\0';
1823
1824 if ((octet1 == '\0') || ((octet1 >> 6) != 2)) { *codepointSize = 2; return codepoint; } // Unexpected sequence
1825
1826 octet2 = text[2];
1827
1828 if ((octet2 == '\0') || ((octet2 >> 6) != 2)) { *codepointSize = 3; return codepoint; } // Unexpected sequence
1829
1830 // [0]xE0 [1]xA0-BF [2]UTF8-tail(x80-BF)
1831 // [0]xE1-EC [1]UTF8-tail [2]UTF8-tail(x80-BF)
1832 // [0]xED [1]x80-9F [2]UTF8-tail(x80-BF)
1833 // [0]xEE-EF [1]UTF8-tail [2]UTF8-tail(x80-BF)
1834
1835 if (((octet == 0xe0) && !((octet1 >= 0xa0) && (octet1 <= 0xbf))) ||
1836 ((octet == 0xed) && !((octet1 >= 0x80) && (octet1 <= 0x9f)))) { *codepointSize = 2; return codepoint; }
1837
1838 if ((octet >= 0xe0) && (octet <= 0xef))
1839 {
1840 codepoint = ((octet & 0xf) << 12) | ((octet1 & 0x3f) << 6) | (octet2 & 0x3f);
1841 *codepointSize = 3;
1842 }
1843 }
1844 else if ((octet & 0xf8) == 0xf0)
1845 {
1846 // Four octets
1847 if (octet > 0xf4) return codepoint;
1848
1849 unsigned char octet1 = text[1];
1850 unsigned char octet2 = '\0';
1851 unsigned char octet3 = '\0';
1852
1853 if ((octet1 == '\0') || ((octet1 >> 6) != 2)) { *codepointSize = 2; return codepoint; } // Unexpected sequence
1854
1855 octet2 = text[2];
1856
1857 if ((octet2 == '\0') || ((octet2 >> 6) != 2)) { *codepointSize = 3; return codepoint; } // Unexpected sequence
1858
1859 octet3 = text[3];
1860
1861 if ((octet3 == '\0') || ((octet3 >> 6) != 2)) { *codepointSize = 4; return codepoint; } // Unexpected sequence
1862
1863 // [0]xF0 [1]x90-BF [2]UTF8-tail [3]UTF8-tail
1864 // [0]xF1-F3 [1]UTF8-tail [2]UTF8-tail [3]UTF8-tail
1865 // [0]xF4 [1]x80-8F [2]UTF8-tail [3]UTF8-tail
1866
1867 if (((octet == 0xf0) && !((octet1 >= 0x90) && (octet1 <= 0xbf))) ||
1868 ((octet == 0xf4) && !((octet1 >= 0x80) && (octet1 <= 0x8f)))) { *codepointSize = 2; return codepoint; } // Unexpected sequence
1869
1870 if (octet >= 0xf0)
1871 {
1872 codepoint = ((octet & 0x7) << 18) | ((octet1 & 0x3f) << 12) | ((octet2 & 0x3f) << 6) | (octet3 & 0x3f);
1873 *codepointSize = 4;
1874 }
1875 }
1876
1877 if (codepoint > 0x10ffff) codepoint = 0x3f; // Codepoints after U+10ffff are invalid
1878
1879 return codepoint;
1880}
