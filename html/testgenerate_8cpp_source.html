15bool verifyData(T_INIFileData const& testData)
16{
17 // compares file contents to expected data
18 std::string line;
19 std::string const& filename = testData.first;
20 T_LineData const& linesExpected = testData.second;
21 size_t lineCount = 0;
22 size_t lineCountExpected = linesExpected.size();
23 std::ifstream fileReadStream(filename);
24 if (fileReadStream.is_open())
25 {
26 while (std::getline(fileReadStream, line))
27 {
28 if (fileReadStream.bad())
29 {
30 return false;
31 }
32 if (lineCount >= lineCountExpected)
33 {
34 std::cout << "Line count exceeds expected!" << std::endl;
35 return false;
36 }
37 std::string const& lineExpected = linesExpected[lineCount++];
38 if (line != lineExpected)
39 {
40 std::cout << "Line " << lineCount << " does not match expected!" << std::endl;
41 std::cout << "Expected: " << lineExpected << std::endl;
42 std::cout << "Is: " << line << std::endl;
43 return false;
44 }
45 }
46 if (lineCount < lineCountExpected)
47 {
48 std::cout << "Line count falls behind expected!" << std::endl;
49 }
50 return lineCount == lineCountExpected;
51 }
52 return false;
53}
